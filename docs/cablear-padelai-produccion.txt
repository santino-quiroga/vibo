// Vibo — fail-open CON CACHE (SDD v2 §1).
//
// Por qué existe este nodo: con /contexto, el llamado a Vibo ya no trae sólo el
// permiso, trae el prompt, las reglas y los precios. Si Vibo no responde y el
// workflow sigue igual (fail-open de v1), el bot contesta SIN instrucciones ni
// precios — cotiza cualquier cosa. Eso es peor que no contestar.
//
// Entonces: se guarda el último contexto bueno y se reusa cuando falla.
//
// OJO: el static data se persiste entre ejecuciones sólo en ejecuciones de
// PRODUCCIÓN. En una ejecución manual de prueba arranca vacío, así que la
// primera prueba manual va a decir sinContexto=true. No es un bug.
// Prompt mínimo para cuando no hay contexto ni cache (Vibo caído en la primera
// ejecución). Antes esto no hacía falta porque la identidad estaba hardcodeada
// en el nodo del agente; al pasar a que la mande Vibo, sin este fallback el bot
// se quedaría sin ninguna instruccion.
//
// Lo importante de este texto no es la identidad: es la prohibicion de inventar
// precios. Un bot sin datos que igual cotiza es peor que uno que dice "ahora te
// confirmo".
const PROMPT_FALLBACK = [
  '## IDENTIDAD',
  'Sos el asistente de un complejo deportivo. Atendes consultas de jugadores por WhatsApp.',
  '',
  '## PERSONA',
  '- Idioma: Espanol rioplatense. Tono natural y cercano. Respuestas cortas.',
  '',
  '## IMPORTANTE',
  '- En este momento NO tenes a mano los precios ni las politicas del complejo.',
  '- Si preguntan por precios, sena o cancelaciones, NO los inventes: deci que en',
  '  un momento alguien del equipo se los confirma.',
].join('\n');

const store = $getWorkflowStaticData('global');
const entrada = $input.first().json || {};

// El contexto sirve si trae promptBase. Un 500, un body vacío o lo que deja
// pasar el "Continue On Fail" no lo traen.
const vivo = typeof entrada.promptBase === 'string' && entrada.promptBase.length > 0;

let ctx = null;
let desdeCache = false;

if (vivo) {
  ctx = entrada;
  store.viboContexto = entrada;
  store.viboContextoAt = new Date().toISOString();
} else if (store.viboContexto) {
  ctx = store.viboContexto;
  desdeCache = true;
}

const money = (n) =>
  typeof n === 'number' ? '$' + n.toLocaleString('es-AR', { maximumFractionDigits: 0 }) : null;

function armarBloque(c) {
  const partes = [];
  const neg = c.negocio || {};

  const datos = [];
  if (neg.nombre) datos.push('- Nombre: ' + neg.nombre);
  if (neg.deporte) datos.push('- Deporte: ' + neg.deporte);
  if (neg.direccion) datos.push('- Direccion: ' + neg.direccion);
  if (neg.telefono) datos.push('- Telefono: ' + neg.telefono);
  if (datos.length) partes.push('## NEGOCIO\n' + datos.join('\n'));

  if (c.tono) partes.push('## TONO\n- ' + c.tono);

  if (Array.isArray(c.canchas) && c.canchas.length) {
    const filas = c.canchas.map((x) => {
      const precio = money(x.precio) || 'sin precio cargado';
      return '- Cancha ' + x.numero + ': ' + precio +
        ' - turnos de ' + x.duracionTurnoMin + ' min' +
        ' - de ' + x.horarioApertura + ' a ' + x.horarioCierre;
    });
    partes.push('## CANCHAS Y PRECIOS\n' + filas.join('\n'));
  }

  const r = c.reglas || {};
  const reglas = [];
  if (typeof r.anticipacionMinimaMin === 'number') {
    const h = r.anticipacionMinimaMin / 60;
    reglas.push('- Anticipacion minima para reservar: ' +
      (Number.isInteger(h) ? h + ' hora(s)' : r.anticipacionMinimaMin + ' minutos'));
  }
  if (r.politicaCancelacion) reglas.push('- Cancelacion: ' + r.politicaCancelacion);

  // OJO: "requiere" NO es un monto, y `false` NO significa "no se cobra sena":
  // significa que NADIE CARGO la politica. Traducir eso a "no se pide sena" es
  // convertir un campo vacio en una afirmacion sobre plata al cliente.
  //
  // Regla de este bloque: solo se afirma lo que esta escrito. Si el complejo
  // realmente no cobra sena, hay que escribirlo en Vibo; asi el bot lo dice
  // porque alguien lo decidio, no porque un campo quedo en blanco.
  if (r.senia && r.senia.requiere && r.senia.detalle) {
    reglas.push('- Sena: ' + r.senia.detalle);
  }
  if (reglas.length) partes.push('## REGLAS DE RESERVA\n' + reglas.join('\n'));

  if (c.faq) partes.push('## PREGUNTAS FRECUENTES\n' + c.faq);

  return partes.join('\n\n');
}

if (!ctx) {
  // Ni contexto vivo ni cache (primera ejecucion con Vibo caido). Se responde
  // igual —fail-open— pero sin bloque: el prompt fijo del nodo sigue sirviendo.
  return [{ json: { puedeResponder: true, desdeCache: false, sinContexto: true,
    promptBase: PROMPT_FALLBACK, bloqueVibo: '' } }];
}

// Fail-open: si el llamado fallo, se responde aunque el cache dijera lo
// contrario. El SDD 4.4 es explicito: es preferible una conversacion de mas que
// cortarle la venta al cliente por una falla de infraestructura nuestra.
const puedeResponder = vivo ? ctx.puedeResponder !== false : true;

return [{
  json: {
    puedeResponder,
    motivo: vivo ? (ctx.motivo === undefined ? null : ctx.motivo) : null,
    desdeCache,
    sinContexto: false,
    cacheAt: store.viboContextoAt || null,
    // La identidad y la persona ahora las manda Vibo, no el nodo del agente.
    promptBase: ctx.promptBase || PROMPT_FALLBACK,
    bloqueVibo: armarBloque(ctx),
  },
}];
